CREATE TABLE user(
   id INT NOT NULL AUTO_INCREMENT,
   first_name VARCHAR(50),
   last_name VARCHAR(50),
   username VARCHAR(50),
   password VARCHAR(50),
   role_id INT,
   disabled TINYINT(1),
   address VARCHAR(100),
   balance Decimal(10,2),
   salary Decimal(10,2),
   PRIMARY KEY ( id ),
   CONSTRAINT fk_Roles FOREIGN KEY (role_id)
REFERENCES roles(id)
);
